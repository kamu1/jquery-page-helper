弹窗   

$(document).on('click","#dialog_btn",function(){
    Helper.dialog({
        content:"确定要取消审核吗？",
        buttons:{
            //方式一
            "确定":function(){
                $(this).remove();
                Helper.tip("点了 确定",1000);
            },
            "取消":function(){
                $(this).remove();
                Helper.tip("点了 取消",2000,"bottom");
            },
            //方式二
            "查看说明":{
                style:{
                    fontWeight:"bold"
                },
                attr:{
                    id:"readBtn"
                },
                className:"read-btn",
                method:function(){
                    Helper.tip("点了 查看说明",3000,"top");
                }
            }
        }
    });
})

或

Helper.selectpicker("#select1,#select2",[
    {noneSelectedText:"区域",width:100},
    {noneSelectedText:"门店",width:200}
]);

或

Helper.selectpicker({
    target:"#select1,#select2",
    option:[
        {noneSelectedText:"区域",width:100},
        {noneSelectedText:"门店",width:200}
    ]
});
